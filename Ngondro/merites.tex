\>\trans{Désormais et jusqu'à ce que j'atteigne le c{\oe}ur de l'Éveil,}\\
\>\stib{bla ma dkon mchog gsum la skyabs su mchi\notsheg:}\\

\>\lex{maître} \>\lex{précieux} \>\lex{trois} \>\lex{en}
\>\lex{je prends refuge}\\
\>\trans{je prends refuge dans le maître et les Trois

  Bouddha, la voie qu'il a montrée (\emph{dharma}) et la
  communauté (monastique et laïque réunies, skt. \emph{sa\.{n}gha})
  qui la parcourt.}\\
\>\stib{da nas bzung ste 'khor ba ma stong bar\notsheg:}\\
\=\lit{da-ne}-----\=\lit{sung-te} \=\lit{khorwa} \quad \

\>\lex{non} \>\lex{vide} \>\lex{tant que}\\
\>\trans{Désormais et tant que les mondes seront peuplés,}\\
\>\stib{ma gyur sems can kun gyi phan bde bsgrub\notsheg:}\\
\>\lit{ma}---\=\lit{gyur} \quad \ \=\lit{semchen} \=\lit{kün}-\=\lit{gyi}
\=\lit{phen}---\=\lit{de} \qquad \=\lit{drub}\\
\>\lex{mère} \>\lex{(ayant) été} \>\lex{êtres} \>\lex{tous} \>\lex{de}
\>\lex{bénéfice} \>\lex{bonheur} \>\lex{accomplir}\\
\>\trans{j'{\oe}uvrerai au bonheur des êtres, mes anciennes mères.}
\end{tabbing}

\rep{À réciter trois fois les mains jointes en prière (ou une fois en
  se prosternant).}
